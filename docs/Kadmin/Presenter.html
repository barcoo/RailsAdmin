def render(view: nil, **options)
  previous_view = @view
  rendered = nil

  begin
    @view = view unless view.nil?
    raise Kadmin::Presenter::NoViewContext if @view.nil?
    rendered = generate(**options)
  ensure
    @view = previous_view
  end

  return rendered
end
